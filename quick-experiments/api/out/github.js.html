const fetch = require('node-fetch');
const accessToken = '44f43b84f20f10834a01d568f6eea603d730f9c1';

/**
 * Retrieve specified resource, if unsppecified return endpoint
 * @param {string} auth - GitHub personal access token
 * @param {string} resource - Resource path, e.g. 'location'
 */

async function getData(auth, resource) {
  const rootEndpoint = 'https://api.github.com';
  const path = '/users/thomasabishop';
  try {
    let response = await fetch(rootEndpoint + path, {
      headers: {
        authorization: `token ${auth}`,
      },
    });
    let data = await response.json();
    let payload;
    !resource ? (payload = data) : (payload = data[resource]);
    console.log(payload);
  } catch (error) {
    console.error(error);
  }
}
